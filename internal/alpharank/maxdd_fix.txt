	// START from initial deposit (not 0!)
	runningBalance := initialDeposit
	peak := initialDeposit
	
	// Loop through closed trades
	for _, trade := range trades {
		netProfit := trade.Profit + trade.Swap + trade.Commission
		runningBalance += netProfit
		
		// Update peak
		if runningBalance > peak {
			peak = runningBalance
		}
		if peak > peakBalance {
			peakBalance = peak
		}
		
		// Calculate DD from closed trades
		if peak > 0 {
			dd := (peak - runningBalance) / peak * 100
			if dd > maxDD {
				maxDD = dd
			}
		}
	}
	
	// CRITICAL: Compare with CURRENT EQUITY (include floating loss!)
	if equity > 0 && peak > 0 {
		currentDD := (peak - equity) / peak * 100
		if currentDD > maxDD {
			maxDD = currentDD
		}
	}
